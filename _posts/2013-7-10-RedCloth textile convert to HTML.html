RedCloth: http://redcloth.org/

{% highlight python %} #!/usr/bin/ruby # # redcloth2html # required packages: ruby libredcloth-ruby # require 'redcloth' if (ARGV.size < 2) then print "redcloth2html \n" print "note: - means stdin or stdout.\n" exit 0 end if ARGV[0] == '-' then infilepath = '/dev/stdin' else infilepath = ARGV[0] end if ARGV[1] == '-' then outfilepath = '/dev/stdout' else outfilepath = ARGV[1] end raw = IO.read infilepath rc = RedCloth.new raw File.open outfilepath, 'w' do |outf| outf.write rc.to_html end {% endhighlight %}
